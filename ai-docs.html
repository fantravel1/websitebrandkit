Overview

The Brand Kit Generator API allows AI assistants (Claude Code, GitHub Copilot, ChatGPT, Cursor) and developers to programmatically generate complete brand asset sets via simple URL parameters.

üí° Perfect for: Automated project setup, CLI tools, CI/CD pipelines, chatbots, and AI-assisted development workflows.

API Endpoint

Base URL:

https://www.websitebrandkit.com/api-generate.html

Method: GET

Response Formats: HTML (default) or JSON

Parameters

Code Examples

JavaScript/Node.js

async function generateBrandKit(config) {
  const params = new URLSearchParams({
    emoji: config.emoji,
    primaryColor: config.primaryColor,
    secondaryColor: config.secondaryColor,
    mainText: config.mainText,
    tagline: config.tagline || '',
    format: 'json'
  });

  const response = await fetch(
    `https://www.websitebrandkit.com/api-generate.html?${params}`
  );

  const data = await response.json();

  // Save images
  for (const [filename, dataUrl] of Object.entries(data.images)) {
    const base64Data = dataUrl.split(',')[1];
    const buffer = Buffer.from(base64Data, 'base64');
    await fs.writeFile(`./public/${filename}`, buffer);
  }

  return data;
}

// Usage

  primaryColor: '#0066FF',

  mainText: 'MyApp',
  tagline: 'Launch Fast'
});

Python

import requests
import base64
from urllib.parse import urlencode

def generate_brand_kit(config):
    params = {
        'emoji': config['emoji'],
        'primaryColor': config['primaryColor'],
        'secondaryColor': config['secondaryColor'],
        'mainText': config['mainText'],
        'tagline': config.get('tagline', ''),
        'format': 'json'
    }

    url = f"https://www.websitebrandkit.com/api-generate.html?{urlencode(params)}"
    response = requests.get(url)
    data = response.json()

    # Save images
    for filename, data_url in data['images'].items():
        base64_data = data_url.split(',')[1]
        image_data = base64.b64decode(base64_data)

        with open(f'./public/{filename}', 'wb') as f:
            f.write(image_data)

    return data

# Usage
generate_brand_kit({
    'emoji': 'üöÄ',
    'primaryColor': '#0066FF',
    'secondaryColor': '#00D4FF',
    'mainText': 'MyApp',
    'tagline': 'Launch Fast'
})
